import matplotlib.pyplot as plt
import pandas as pd

def create_stacked_bar_chart(data, date_column, status_column, title='Yearly Distribution by Status', xlabel='Year', ylabel='Count', legend_labels=['Category 1', 'Category 2'], figsize=(14, 8)):
    """
    Generates a stacked bar chart for yearly data distribution based on a binary status, ensuring year labels are integers.

    Parameters:
    - data (DataFrame): The pandas DataFrame containing the dataset.
    - date_column (str): The column name in the DataFrame containing the date information.
    - status_column (str): The column name indicating the binary status for stacking.
    - title (str): Title of the chart.
    - xlabel (str): Label for the x-axis.
    - ylabel (str): Label for the y-axis.
    - legend_labels (list): Labels for the legend, corresponding to binary status.
    - figsize (tuple): Figure size in the form of (width, height).
    """

    # Copying the DataFrame to avoid the warning
    df_copy = data.copy()

    # Extracting year and directly converting to string
    df_copy['year_str'] = pd.to_datetime(df_copy[date_column], errors='coerce').dt.year.astype('str')

    # Grouping data by the year string and binary status, counting occurrences
    yearly_data = df_copy.groupby(['year_str', status_column])[date_column].count().unstack(fill_value=0)

    # Plotting
    fig, ax = plt.subplots(figsize=figsize)
    yearly_data.plot(kind='bar', stacked=True, ax=ax, color=['#1f77b4', '#ff7f0e'])

    # Setting titles and labels
    ax.set_title(title, fontsize=16)
    ax.set_xlabel(xlabel, fontsize=14)
    ax.set_ylabel(ylabel, fontsize=14)
    ax.legend(legend_labels, title=status_column)

    # Adjusting x-axis to display years as integers, handling 'nan' and converting valid years to integers
    ax.set_xticklabels([label if label == 'nan' else int(float(label)) for label in yearly_data.index])

    # Formatting the plot for better readability
    plt.xticks(rotation=45)
    plt.tight_layout()

    # Show plot
    plt.show()
